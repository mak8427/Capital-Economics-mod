# Maintainer Note:
# - Purpose: CapEcon_corporate_events.txt defines event chains and event outcomes used by the Pantalone_Economics feature set.
# - Reasoning: keeping this logic in a dedicated file isolates balancing and avoids accidental vanilla overlap.
# - Maintenance: when renaming keys/scopes, update cross-file references and localization entries in the same change.

namespace = cap_econ_corp

cap_econ_corp.1 = {
    type = country_event
    hidden = yes
    title = cap_econ_corp.1.t

    trigger = {
        is_subject_type = international_company
    }

    immediate = {
        overlord ?= { save_scope_as = cap_econ_corp_overlord }
        if = {
            limit = {
                scope:cap_econ_corp_overlord = {
                    within_diplomatic_range = root
                }
            }
            set_variable = { name = cap_econ_corporate_build_cost value = 500 }
            # Spend available gold in 500 chunks; each chunk triggers one build attempt.
            while = {
                limit = { gold >= var:cap_econ_corporate_build_cost }
                count = 100
                if = {
                    # First build always targets the company itself to guarantee local growth.
                    limit = { NOT = { has_variable = cap_econ_corp_first_built } }
                    set_variable = { name = cap_econ_corp_first_built value = 1 }
                    save_scope_as = cap_econ_corp_target
                }
                else = {
                    # Later builds are distributed to valid partners; fall back to overlord if none are valid.
                    random_country = {
                        limit = {
                            religion = religion:pantalonism
                            within_diplomatic_range = root
                            NOT = { this = scope:cap_econ_corp_overlord }
                        }
                        save_scope_as = cap_econ_corp_target
                    }
                    if = {
                        limit = { NOT = { exists = scope:cap_econ_corp_target } }
                        scope:cap_econ_corp_overlord ?= { save_scope_as = cap_econ_corp_target }
                    }
                }
                scope:cap_econ_corp_target ?= {
                    cap_econ_corporate_build_once = yes
                }
            }
            remove_variable = cap_econ_corporate_build_cost
            remove_variable = cap_econ_corp_first_built
        }
    }

    option = {
        name = cap_econ_corp.1.a
    }
}

cap_econ_corp.2 = {
    type = country_event
    hidden = yes
    title = cap_econ_corp.2.t

    trigger = {
        is_subject_type = international_company
    }

    immediate = {
        overlord ?= { save_scope_as = cap_econ_corp_overlord }
        if = {
            limit = {
                scope:cap_econ_corp_overlord = {
                    within_diplomatic_range = root
                    giving_scripted_relation = {
                        type = relation_type:cap_econ_extraction_rights
                        target = root
                    }
                }
            }
            set_variable = { name = cap_econ_corporate_build_cost value = 500 }
            # Same spending loop as event .1, but partner selection is restricted to extraction-rights holders.
            while = {
                limit = { gold >= var:cap_econ_corporate_build_cost }
                count = 100
                if = {
                    limit = { NOT = { has_variable = cap_econ_corp_first_built } }
                    set_variable = { name = cap_econ_corp_first_built value = 1 }
                    save_scope_as = cap_econ_corp_target
                }
                else = {
                    random_country = {
                        limit = {
                            giving_scripted_relation = {
                                type = relation_type:cap_econ_extraction_rights
                                target = root
                            }
                            within_diplomatic_range = root
                            NOT = { this = scope:cap_econ_corp_overlord }
                        }
                        save_scope_as = cap_econ_corp_target
                    }
                    if = {
                        limit = { NOT = { exists = scope:cap_econ_corp_target } }
                        scope:cap_econ_corp_overlord ?= { save_scope_as = cap_econ_corp_target }
                    }
                }
                scope:cap_econ_corp_target ?= {
                    cap_econ_corporate_build_once = yes
                }
            }
            remove_variable = cap_econ_corporate_build_cost
            remove_variable = cap_econ_corp_first_built
        }
    }

    option = {
        name = cap_econ_corp.1.a
    }
}
