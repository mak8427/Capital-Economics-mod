cap_econ_send_pantalone = {
    type = diplomacy
    relation_type = oneway
    break_on_war = no
    category = {
        offer = CATEGORY_HOSTILE_ACTIONS
        request = CATEGORY_HOSTILE_ACTIONS
        cancellation = CATEGORY_HOSTILE_ACTIONS
        break = CATEGORY_HOSTILE_ACTIONS
    }

    relation_type_for_ai = hostile

    break_enabled = {
        scope:recipient = { religion = religion:pantalonism }
    }


    visible = {
        scope:actor = {
            AND = {
                within_diplomatic_range = scope:recipient
                scope:actor = { religion = religion:pantalonism }
            }
        }
    }
    offer_enabled = {
        scope:actor = {
            NOT = {
                giving_scripted_relation_of_type = relation_type:cap_econ_send_pantalone
            }
        }
    }

    offer_effect = {
        scope:recipient = {
            trigger_event_non_silently = cap_econ.5
            set_variable = { name = cap_econ_pantalone_sender value = scope:actor }
        }
    }
    cancel_effect = {
        scope:recipient = {
            remove_variable = cap_econ_pantalone_sender
        }
    }
    break_effect = {
        scope:recipient = {
            remove_variable = cap_econ_pantalone_sender
        }
    }
    expire_effect = {
        scope:recipient = {
            remove_variable = cap_econ_pantalone_sender
        }
    }

    wants_to_receive = {
        add = {
            desc = "DIPLOREASON_GOOD_FOR_US"
            value = 100
        }
    }



    wants_to_give = {
        add = 500
        add = {
            desc = "game_concept_rival"
            if = { limit = { scope:recipient = { is_rival_of = scope:actor } } }
            value = 200
        }

        add = {
            desc = "IS_WAR_ENEMY"
            if = { limit = { scope:recipient = { is_at_war_with = scope:actor } } }
            value = 200
        }

        # Prefer countries we dislike
        subtract = {
            desc = "game_concept_opinion"
            scope:actor = { value = "opinion(scope:recipient)" }
        }
    }

}
