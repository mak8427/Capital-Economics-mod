international_company = {
	type = building
	color = subject_trade_company
	level = 1
	subject_pays = subject_pays_international_company

	join_defensive_wars_always = {
		scope:actor = {
			is_subject_of = scope:recipient
		}
	}
	has_overlords_ruler = no

	subject_can_cancel = no
	overlord_can_cancel = no

	fleet_basing_rights = yes
	has_limited_diplomacy = yes

	can_change_rank = no
	can_change_heir_selection = no

	strength_vs_overlord = -0.1
	diplomatic_capacity_cost_scale = 0.5

	government = republic

	minimum_opinion_for_offer = 100

	great_power_score_transfer = 0.75

	institution_spread_to_overlord = monthly_institution_spread_severe
	institution_spread_to_subject = monthly_institution_spread_severe

	overlord_modifier = { }

	subject_modifier = {
		global_production_efficiency = tiny_production_efficiency_bonus
		trade_efficiency = tiny_trade_efficiency_bonus
	}

	creation_visible = {
		always = yes
	}

	visible_through_diplomacy = {
		always = no
	}

	visible_through_treaty = {
		always = no
	}

	on_enable = {
		create_relation = {
			type = relation_type:cap_econ_extraction_rights
			first = scope:future_overlord
			second = root
		}
		add_gold = 2000
	}

	on_disable = {
		remove_relation = {
			type = relation_type:cap_econ_extraction_rights
			first = scope:former_overlord
			second = root
		}
	}

	on_monthly = {
		if = {
			limit = { current_month = 2 }
			if = {
				limit = { religion = religion:pantalonism }
				trigger_event_non_silently = cap_econ_corp.1
			}
			else = {
				trigger_event_non_silently = cap_econ_corp.2
			}
		}
	}
}
