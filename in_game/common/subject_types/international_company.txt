# Maintainer Note:
# - Purpose: international_company.txt defines subject type rules used by the Pantalone_Economics feature set.
# - Reasoning: keeping this logic in a dedicated file isolates balancing and avoids accidental vanilla overlap.
# - Maintenance: when renaming keys/scopes, update cross-file references and localization entries in the same change.

international_company = {
	type = building
	color = subject_trade_company
	level = 1
	subject_pays = subject_pays_international_company

	join_defensive_wars_always = {
		scope:actor = {
			is_subject_of = scope:recipient
		}
	}
	has_overlords_ruler = no

	subject_can_cancel = no
	overlord_can_cancel = no

	fleet_basing_rights = yes
	has_limited_diplomacy = yes

	can_change_rank = no
	can_change_heir_selection = no

	strength_vs_overlord = -0.1
	diplomatic_capacity_cost_scale = 0.5

	government = company

	minimum_opinion_for_offer = 100

	great_power_score_transfer = 0.75

	institution_spread_to_overlord = monthly_institution_spread_severe
	institution_spread_to_subject = monthly_institution_spread_severe

	overlord_modifier = { }

	subject_modifier = {
		global_production_efficiency = tiny_production_efficiency_bonus
		trade_efficiency = tiny_trade_efficiency_bonus
	}

	creation_visible = {
		always = yes
	}

	visible_through_diplomacy = {
		always = no
	}

	visible_through_treaty = {
		always = no
	}

	on_enable = {
		# Enforce deterministic identity for newly created companies.
		# Without this block, release paths can fall back to generic rank/title and reform defaults.
		change_country_type = building
		change_government_type = government_type:company
		remove_current_major_government_reform = yes
		# Clear inherited reforms first so the company always starts from the intended reform package.
		save_temporary_scope_as = cap_econ_international_company_scope
		every_current_reforms = {
			scope:cap_econ_international_company_scope = {
				remove_reform = prev
			}
		}
		add_reform = government_reform:pantalone_company
		add_reform = government_reform:council_of_burghers_company
		create_relation = {
			type = relation_type:cap_econ_extraction_rights
			first = scope:future_overlord
			second = root
		}
		add_gold = 2000
	}

	on_disable = {
		remove_relation = {
			type = relation_type:cap_econ_extraction_rights
			first = scope:former_overlord
			second = root
		}
	}

	on_monthly = {
		if = {
			limit = { current_month = 2 }
			if = {
				limit = { religion = religion:pantalonism }
				trigger_event_non_silently = cap_econ_corp.1
			}
			else = {
				trigger_event_non_silently = cap_econ_corp.2
			}
		}
	}
}
