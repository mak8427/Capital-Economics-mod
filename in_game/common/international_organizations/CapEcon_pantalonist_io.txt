# Maintainer Note:
# - Purpose: CapEcon_pantalonist_io.txt defines international organization setup used by the Pantalone_Economics feature set.
# - Reasoning: keeping this logic in a dedicated file isolates balancing and avoids accidental vanilla overlap.
# - Maintenance: when renaming keys/scopes, update cross-file references and localization entries in the same change.

cap_econ_pantalonist_io = {
	has_target = no
	unique = yes
	expel_members_who_are_targets_of_other_members = no

	has_leader_country = yes
	leader_type = character
	leader_change_trigger_type = none
	leader_change_method = vote
	leadership_election_resolution = cap_econ_pantalonist_io_election
	disband_if_no_leader = no
	show_on_diplomatic_map = yes
	leader_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
	has_parliament = yes
	parliament_type = no_organization_parliament
	resolution_widget = parliament
	laws = {
		pantalone_stability_law = no_pantalone_stability_policy
		pantalone_production_law = no_pantalone_production_policy
		pantalone_imperial_diet_law = no_pantalone_imperial_diet_policy
		pantalone_contribution_law = no_pantalone_contribution_policy
		pantalone_estate_contribution_law = no_pantalone_estate_contribution_policy
		pantalone_leader_payment_law = pantalone_leader_exempt_policy
		pantalone_redistribution_law = pantalone_dividends_leader_policy
		pantalone_defense_law = no_pantalone_defense_policy
	}
	on_creation = {
		international_organization_unlock_law_effect = { type = pantalone_stability_law }
		international_organization_unlock_law_effect = { type = pantalone_production_law }
		international_organization_unlock_law_effect = { type = pantalone_imperial_diet_law }
		international_organization_unlock_law_effect = { type = pantalone_contribution_law }
		international_organization_unlock_law_effect = { type = pantalone_estate_contribution_law }
		international_organization_unlock_law_effect = { type = pantalone_leader_payment_law }
		international_organization_unlock_law_effect = { type = pantalone_redistribution_law }
		international_organization_unlock_law_effect = { type = pantalone_defense_law }
		set_variable = cap_econ_io_diet_migrated
	}


	leader = {
		leader_country ?= {
			ruler ?= { add_to_list = leaders }
		}
	}

	variables = {
		religion = {
			format = "RELIGION_FORMAT"
			start = religion:pantalonism
		}
		pantalone_authority = {
			format = "PANTALONE_AUTHORITY_DISPLAY"
			change_format = "VARIABLE_CHANGE_FORMAT"
			min = 0
			max = 100
			start = 0

			monthly_change = {
				add = {
					add = {
						desc = "PANTALONE_AUTHORITY_BASE"
						value = 0.05
					}
					add = {
						desc = "PANTALONE_AUTHORITY_MEMBERS"
						value = total_members
						multiply = 0.05
					}
					add = {
						desc = "PANTALONE_AUTHORITY_LEADER_STABILITY"
						value = leader_country.stability
						multiply = 0.01
					}
				}
			}
		}
		contribution_modifier = {
			format = "CAP_ECON_DIVIDENDS_FORMAT"
			start = 20
		}
		estate_contribution_modifier = {
			format = "CAP_ECON_DIVIDENDS_FORMAT"
			start = 10
		}
	}

	create_visible_trigger = { always = no }
	invite_visible_trigger = { always = no }

	can_join_trigger = {
		religion = religion:pantalonism
	}
	can_initiate_policy_votes = {
		is_leader_of_international_organization = scope:recipient
	}
	can_vote_in_parliament = {
		is_member_of_international_organization = scope:recipient
	}
	can_leave_trigger = { always = no }
	auto_leave_trigger = {
		NOT = { religion = religion:pantalonism }
	}
	can_lead_trigger = {
		is_subject = no
		religion = religion:pantalonism
	}

	monthly_effect = {
		# Existing saves created before the diet law was added should keep a usable parliament setup.
		if = {
			limit = {
				NOT = { has_variable = cap_econ_io_diet_migrated }
			}
			international_organization_unlock_law_effect = { type = pantalone_imperial_diet_law }
			if = {
				limit = {
					NOT = {
						OR = {
							international_organization_has_policy = policy:pantalone_early_imperial_diet_policy
							international_organization_has_policy = policy:pantalone_bi_camerial_imperial_diet_policy
							international_organization_has_policy = policy:pantalone_tri_camerial_imperial_diet_policy
						}
					}
				}
				add_policy_to_international_organization = policy:pantalone_early_imperial_diet_policy
			}
			set_variable = cap_econ_io_diet_migrated
		}

		if = {
			limit = {
				international_organization_has_policy = policy:pantalone_stability_policy
			}
			every_international_organization_member = {
				add_stability = 0.10
			}
		}
	}
	ai_desire_to_join = { }
	ai_desire_to_allow_new_member = { }

	payments_implemented = {
		pantalone_contribution
	}

}
