cap_econ_pantalonist_io = {
	has_target = no
	unique = yes
	expel_members_who_are_targets_of_other_members = no

	has_leader_country = yes
	leader_type = country
	leader_change_trigger_type = none
	leader_change_method = none
	disband_if_no_leader = no
	show_on_diplomatic_map = yes
	leader_color = define:NMapColors|INTERNATIONAL_ORGANIZATION_LEADER_COLOR
	has_parliament = yes
	parliament_type = hre_court_assembly
	resolution_widget = parliament
	laws = {
		pantalone_stability_law = no_pantalone_stability_policy
		pantalone_production_law = no_pantalone_production_policy
	}
	on_creation = {
		international_organization_unlock_law_effect = { type = pantalone_stability_law }
		international_organization_unlock_law_effect = { type = pantalone_production_law }
	}


	leader = {
		leader_country ?= {
			add_to_list = leaders
		}
	}

	variables = {
		religion = {
			format = "RELIGION_FORMAT"
			start = religion:pantalonism
		}
		pantalone_authority = {
			format = "PANTALONE_AUTHORITY_DISPLAY"
			change_format = "VARIABLE_CHANGE_FORMAT"
			min = 0
			max = 100
			start = 0

			monthly_change = {
				add = {
					add = {
						desc = "PANTALONE_AUTHORITY_BASE"
						value = 0.05
					}
					add = {
						desc = "PANTALONE_AUTHORITY_MEMBERS"
						value = total_members
						multiply = 0.002
					}
					add = {
						desc = "PANTALONE_AUTHORITY_LEADER_STABILITY"
						value = leader_country.stability
						multiply = 0.01
					}
				}
			}
		}
	}

	create_visible_trigger = { always = no }
	invite_visible_trigger = { always = no }

	can_join_trigger = {
		religion = religion:pantalonism
	}
	can_leave_trigger = { always = no }
	auto_leave_trigger = {
		NOT = { religion = religion:pantalonism }
	}
	can_lead_trigger = {
		is_subject = no
		religion = religion:pantalonism
	}

	monthly_effect = {
		if = {
			limit = {
				international_organization_has_policy = policy:pantalone_stability_policy
			}
			every_international_organization_member = {
				add_stability = 0.10
			}
		}
	}
	ai_desire_to_join = { }
	ai_desire_to_allow_new_member = { }

	payments_implemented = {
		pantalone_contribution
	}

}
